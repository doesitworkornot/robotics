alias
#Присваивает созданные вами названия длинным командам,
# которые вы не можете запомнить.
# Введите alias длинная_команда короткая_команда.

history
#Отображает все недавно введённые вами команды.
# Кроме того, вы можете переключаться между недавними с помощью клавиш 
#«Вверх» и «Вниз». Если вы не хотите, чтобы введённая
# вами команда была записана, поставьте перед ней пробел.
# Так: ваша_команда.

whatis
#Отображает краткое описание какой‑либо программы.
# Введите команду и название программы: whatis имя_пакета.

sudo su
#Благодаря этой комбинации все введённые вами команды будут
# исполняться от имени суперпользователя, пока вы не закроете терминал.
# Применяйте её, если вам нужно выполнить много команд с правами администратора.

chown
#Изменяет владельца файла или каталога.
#Следует выполнять с правами суперпользователя.
#Например, chown user:group ваш_файл изменит владельца и группу файла
# на заданные.

find
#Поиск файлов по определённым критериям, 
#таким как имя, тип, размер, владелец, дата создания и модификации.

locate
#Поиск файлов и папок, чьи названия подходят запросу, 
#и отображение их путей в файловой системе.

lsblk
#Эта команда демонстрирует, какие диски есть в вашей системе
# и на какие разделы они поделены.
# Также она отображает имена ваших разделов и накопителей в формате 
#sda1, sda2 и так далее.

free
#Показывает объём доступной и занятой оперативной памяти.

whoami
#Называет имя пользователя. 
#Полезно, если вы забыли, под каким логином зашли в систему.
